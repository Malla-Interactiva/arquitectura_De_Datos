Estructura de las tablas creadas en comunas_norm.db (SQLite):

-- ================================================ -- Tabla 1: comunas normalizadas y únicas -- ================================================ CREATE TABLE IF NOT EXISTS COMUNAS_NORM ( id INTEGER PRIMARY KEY AUTOINCREMENT, nombre_comuna TEXT NOT NULL UNIQUE, -- UNIQUE evita duplicados fecha_insercion TEXT NOT NULL ); -- ================================================ -- Tabla 2: historial de cada transformación ETL -- ================================================ CREATE TABLE IF NOT EXISTS PROCESO_LOG ( id INTEGER PRIMARY KEY AUTOINCREMENT, sesion_id TEXT NOT NULL, -- timestamp del procesamiento linea_num INTEGER, -- n° de línea en el archivo original valor_orig TEXT, -- valor antes de normalizar valor_norm TEXT, -- valor después de normalizar estado TEXT, -- OK | MODIFICADO | DUPLICADO fecha TEXT NOT NULL );

Consultas de ejemplo:

-- Ver todas las comunas ordenadas alfabéticamente SELECT * FROM COMUNAS_NORM ORDER BY nombre_comuna; -- Contar registros por estado SELECT estado, COUNT(*) FROM PROCESO_LOG GROUP BY estado; -- Ver solo los valores que fueron modificados SELECT valor_orig, valor_norm FROM PROCESO_LOG WHERE estado = 'MODIFICADO';
